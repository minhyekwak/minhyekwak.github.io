I made an web page by utilizing pokemon API. When the page is loaded, input box and button are in top of the page.
If user input the ID number of pokemon and press "GO" button, then each pokemon's name, image, weight, and height show up below.

"use strict";

var search = function search(event) {
event.preventDefault();
var pokenumber = document.getElementById("pokenumber").value;
var baseURL = "https://pokeapi.co/api/v2/pokemon/";
$.get(baseURL + pokenumber, function (pokemon) {
document.getElementById("pokename").innerHTML = pokemon.name;
document.getElementById("pokeimage").src = pokemon.sprites.front_default;
document.getElementById("weight").innerHTML = pokemon.weight;
document.getElementById("height").innerHTML = pokemon.height;
});
};

document.getElementById("pokeform").addEventListener('submit', search);
